# To enter interactive mode:
docker-compose exec db psql -U postgres -d moview_db

# Once inside psql, you can run:
\dt                    # List all tables
\d users              # Describe users table structure
\d reviews            # Describe reviews table structure

SELECT * FROM users;   # View all users
SELECT * FROM reviews; # View all reviews

# Advanced queries:
SELECT u.username, COUNT(r.id) as review_count 
FROM users u 
LEFT JOIN reviews r ON u.id = r.user_id 
GROUP BY u.username;

# To exit: \q
